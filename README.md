            if(s[i] !== s[end - i + start])
                return false;
        }

        }

        sa.push(root)
        ans.push(root.val)
    }
    while(sa.length > 0 || sb.length > 0) {
        if(sa.length <= 0) {
            sa = sb
            sb = []
        }
        t = null
        while(sa.length > 0) {
            let p = sa.shift()
            if(p.left) {
                t = p.left
                sb.push(t)
            }
            if(p.right) {
                t = p.right
                sb.push(t)
            }
        }
        if(t) {
            ans.push(t.val)
        }
    }
    return ans
};
